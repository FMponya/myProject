# Query to create Employee Table, Projects Table, Department Table and Responsibility Table

CREATE TABLE Employee ( Emp_No VARCHAR(20), Emp_Name VARCHAR(70), B_Date VARCHAR(30), Title VARCHAR(290), Salary INT(10), Department_No VARCHAR(15) );

CREATE TABLE Projects ( Proj_No VARCHAR(15), Proj_Name VARCHAR(265), Budget CHAR (30), Department_No VARCHAR(270) );

CREATE TABLE Department ( Department_No VARCHAR(18), Department_Name VARCHAR(55) );

CREATE TABLE Schedule ( Emp_No VARCHAR(19), Proj_No VARCHAR(20), Responsibility VARCHAR(300), Hours INT );

# 3. An SQL query that returns the project number and name for projects with a budget greater than MK500, 000
SELECT Proj_No, Proj_Name FROM Projects WHERE Budget > '500000';

# 4. An SQL query that returns all works on records where hours worked is less than 10. SELECT * FROM Schedule WHERE Hours < 10;

# 5. An SQL query that returns the employees (number and name only) who receive less than MK350, 000. SELECT Emp_No, Emp_Name FROM Employee WHERE Salary < 350000;

# 6. An SQL query that returns the employees (name only) in department 'D1' ordered by decreasing salary. SELECT Emp_Name FROM Employee WHERE Department_No = 'D1' ORDER BY Salary DESC;

# 7. An SQL query that returns the departments (all fields) ordered by ascending department name. SELECT * FROM Department ORDER BY Department_Name ASC;

# 8. An SQL query that returns the employee name, department name, and employee title. SELECT Employee.Emp_Name, Department.Department_Name, Employee.Title FROM Employee INNER JOIN Department ON Employee.Department_No=Department.Department_No;

# 9. An SQL query that returns the project name, hours worked, and project number for all works on records where hours > 15-09-1995
SELECT Projects.Proj_Name, Schedule.Hours, Schedule.Proj_No FROM Projects INNER JOIN Schedule ON Projects.Proj_No=Schedule.Proj_No WHERE Schedule.Hours>10;
